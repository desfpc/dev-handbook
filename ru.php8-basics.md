## –û–±–ª–∞—Å—Ç–∏ –≤–∏–¥–∏–º–æ—Å—Ç–∏ –∏ –∑–∞–º—ã–∫–∞–Ω–∏—è

### –ö–ª—é—á–µ–≤—ã–µ –º–æ–º–µ–Ω—Ç—ã

### –ü–æ–¥–≤–æ–¥–Ω—ã–µ –∫–∞–º–Ω–∏

## –ú–∞—Å—Å–∏–≤—ã –∏ –∏—Ö –æ—Å–æ–±–µ–Ω–Ω–æ—Å—Ç–∏

–ú–∞—Å—Å–∏–≤—ã –≤ PHP ‚Äî –º–æ—â–Ω—ã–π –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç, –ø–æ–¥–¥–µ—Ä–∂–∏–≤–∞—é—â–∏–π —á–∏—Å–ª–æ–≤—ã–µ –∏ –∞—Å—Å–æ—Ü–∏–∞—Ç–∏–≤–Ω—ã–µ –∫–ª—é—á–∏, –Ω–æ –∏—Ö –ø–æ–≤–µ–¥–µ–Ω–∏–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –Ω–µ–æ—á–µ–≤–∏–¥–Ω—ã–º. –ù–µ–ø—Ä–∞–≤–∏–ª—å–Ω–æ–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ –º–∞—Å—Å–∏–≤–æ–≤ –∏–ª–∏ —Ñ—É–Ω–∫—Ü–∏–π –¥–ª—è —Ä–∞–±–æ—Ç—ã —Å –Ω–∏–º–∏ —á–∞—Å—Ç–æ –ø—Ä–∏–≤–æ–¥–∏—Ç –∫ –æ—à–∏–±–∫–∞–º, –æ—Å–æ–±–µ–Ω–Ω–æ —É –Ω–∞—á–∏–Ω–∞—é—â–∏—Ö —Ä–∞–∑—Ä–∞–±–æ—Ç—á–∏–∫–æ–≤. –≠—Ç–æ—Ç —Ä–∞–∑–¥–µ–ª —Ä–∞–∑–±–∏—Ä–∞–µ—Ç —Ç–∏–ø–∏—á–Ω—ã–µ –æ—à–∏–±–∫–∏ –∏ —Ñ—É–Ω–∫—Ü–∏–∏, –≤ –∫–æ—Ç–æ—Ä—ã—Ö –ø—É—Ç–∞—é—Ç—Å—è, –∞ —Ç–∞–∫–∂–µ –¥–∞—ë—Ç —Å–æ–≤–µ—Ç—ã –¥–ª—è —Å–æ–±–µ—Å–µ–¥–æ–≤–∞–Ω–∏–π.

### –ö–ª—é—á–µ–≤—ã–µ –º–æ–º–µ–Ω—Ç—ã
- **–ß–∏—Å–ª–æ–≤—ã–µ –∏ –∞—Å—Å–æ—Ü–∏–∞—Ç–∏–≤–Ω—ã–µ –º–∞—Å—Å–∏–≤—ã**: PHP –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –ø—Ä–∏—Å–≤–∞–∏–≤–∞–µ—Ç —á–∏—Å–ª–æ–≤—ã–µ –∫–ª—é—á–∏, –Ω–æ –∏—Ö –º–æ–∂–Ω–æ —Å–º–µ—à–∏–≤–∞—Ç—å —Å–æ —Å—Ç—Ä–æ–∫–æ–≤—ã–º–∏, —á—Ç–æ –∏–Ω–æ–≥–¥–∞ –≤—ã–∑—ã–≤–∞–µ—Ç –ø—É—Ç–∞–Ω–∏—Ü—É.
  ```php

  ```
- **–û–ø–µ—Ä–∞—Ç–æ—Ä—ã –æ–±—ä–µ–¥–∏–Ω–µ–Ω–∏—è**: –û–ø–µ—Ä–∞—Ç–æ—Ä `+` –∏ —Ñ—É–Ω–∫—Ü–∏—è `array_merge()` –≤–µ–¥—É—Ç —Å–µ–±—è –ø–æ-—Ä–∞–∑–Ω–æ–º—É.

  ```
- **–£–¥–∞–ª–µ–Ω–∏–µ —ç–ª–µ–º–µ–Ω—Ç–æ–≤**: `unset()` –Ω–µ –ø–µ—Ä–µ–∏–Ω–¥–µ–∫—Å–∏—Ä—É–µ—Ç —á–∏—Å–ª–æ–≤—ã–µ –∫–ª—é—á–∏, —á—Ç–æ –º–æ–∂–µ—Ç –Ω–∞—Ä—É—à–∏—Ç—å –ª–æ–≥–∏–∫—É.

### –¢–∏–ø–∏—á–Ω—ã–µ –æ—à–∏–±–∫–∏
- **–°–º–µ—à–∏–≤–∞–Ω–∏–µ –∫–ª—é—á–µ–π**: –ß–∏—Å–ª–æ–≤—ã–µ –∏ —Å—Ç—Ä–æ–∫–æ–≤—ã–µ –∫–ª—é—á–∏ –º–æ–≥—É—Ç –∫–æ–Ω—Ñ–ª–∏–∫—Ç–æ–≤–∞—Ç—å, –æ—Å–æ–±–µ–Ω–Ω–æ –µ—Å–ª–∏ —Å—Ç—Ä–æ–∫–∞ –∏–Ω—Ç–µ—Ä–ø—Ä–µ—Ç–∏—Ä—É–µ—Ç—Å—è –∫–∞–∫ —á–∏—Å–ª–æ.

  ```
- **–ù–µ–≤–µ—Ä–Ω–æ–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ `unset` –≤ —Ü–∏–∫–ª–∞—Ö**: –£–¥–∞–ª–µ–Ω–∏–µ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –≤ —Ü–∏–∫–ª–µ `foreach` –º–æ–∂–µ—Ç –ø—Ä–∏–≤–µ—Å—Ç–∏ –∫ –ø—Ä–æ–ø—É—Å–∫—É —ç–ª–µ–º–µ–Ω—Ç–æ–≤.

  **–û—à–∏–±–∫–∞**: –ï—Å–ª–∏ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å `foreach` —Å –∏–∑–º–µ–Ω–µ–Ω–∏–µ–º –º–∞—Å—Å–∏–≤–∞, –∏—Ç–µ—Ä–∞—Ç–æ—Ä –º–æ–∂–µ—Ç –ø—Ä–æ–ø—É—Å—Ç–∏—Ç—å —ç–ª–µ–º–µ–Ω—Ç—ã. –ò—Å–ø–æ–ª—å–∑—É–π—Ç–µ `array_filter` –∏–ª–∏ —Ü–∏–∫–ª `for` –¥–ª—è –±–µ–∑–æ–ø–∞—Å–Ω–æ–≥–æ —É–¥–∞–ª–µ–Ω–∏—è.

### –§—É–Ω–∫—Ü–∏–∏, –≤ –∫–æ—Ç–æ—Ä—ã—Ö –ø—É—Ç–∞—é—Ç—Å—è
- **`array_merge()`**: –û–±—ä–µ–¥–∏–Ω—è–µ—Ç –º–∞—Å—Å–∏–≤—ã, –ø–µ—Ä–µ–∑–∞–ø–∏—Å—ã–≤–∞—è –∑–Ω–∞—á–µ–Ω–∏—è –æ–¥–∏–Ω–∞–∫–æ–≤—ã—Ö –∫–ª—é—á–µ–π. –î–ª—è —á–∏—Å–ª–æ–≤—ã—Ö –∫–ª—é—á–µ–π —Å–æ–∑–¥–∞—ë—Ç –Ω–æ–≤—ã–µ –∏–Ω–¥–µ–∫—Å—ã.

    if ($left < $right) {
        $pivotIndex = partition($array, $left, $right);
        quickSort($array, $left, $pivotIndex - 1);
        quickSort($array, $pivotIndex + 1, $right);
    }
}

function partition(&$array, $left, $right) {
    $pivot = $array[$right];
    $i = $left - 1;
    for ($j = $left; $j < $right; $j++) {
        if ($array[$j] <= $pivot) {
            $i++;
            [$array[$i], $array[$j]] = [$array[$j], $array[$i]];
        }
    }
    [$array[$i + 1], $array[$right]] = [$array[$right], $array[$i + 1]];
    return $i + 1;
}
